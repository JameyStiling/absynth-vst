cmake_minimum_required(VERSION 3.22)

# 1. Initialize the Project
project(AbsynthSynth VERSION 0.0.1)

# 2. Include JUCE
add_subdirectory(JUCE)

# 3. Create the Plugin Target
juce_add_plugin(AbsynthSynth
    COMPANY_NAME "Sherd Audio"
    IS_SYNTH TRUE                 # Tells DAW this is an instrument
    NEEDS_MIDI_INPUT TRUE         # So you can play it with a keyboard
    PRODUCT_NAME "Lattice"
    PLIST_TO_MERGE "<dict><key>NSAppTransportSecurity</key><dict><key>NSAllowsArbitraryLoads</key><true/><key>NSAllowsLocalNetworking</key><true/></dict></dict>"
    FORMATS VST3 Standalone       # Build format for VST3 and Standalone app
)

# Ignore VST3 parameter ID warning since this plugin has no parameters
add_definitions(-DJUCE_IGNORE_VST3_MISMATCHED_PARAMETER_ID_WARNING)

# 4. Link your C++ files
target_sources(AbsynthSynth PRIVATE
    Source/PluginProcessor.cpp
    Source/PluginEditor.cpp
)

# 5. Link JUCE Modules (The Web View is essential here!)
target_link_libraries(AbsynthSynth PRIVATE
    juce::juce_audio_utils
    juce::juce_gui_extra          # This holds the WebBridge logic
    juce::juce_dsp                # subtractive synth math
)

# 6. Bypass macOS App Transport Security for the Standalone App
if (APPLE)
    add_custom_command(TARGET AbsynthSynth_Standalone POST_BUILD
        COMMAND /usr/libexec/PlistBuddy -c "Add :NSAppTransportSecurity dict" "$<TARGET_FILE_DIR:AbsynthSynth_Standalone>/../Info.plist" || true
        COMMAND /usr/libexec/PlistBuddy -c "Add :NSAppTransportSecurity:NSAllowsArbitraryLoads bool true" "$<TARGET_FILE_DIR:AbsynthSynth_Standalone>/../Info.plist" || true
        COMMAND /usr/libexec/PlistBuddy -c "Add :NSAppTransportSecurity:NSAllowsLocalNetworking bool true" "$<TARGET_FILE_DIR:AbsynthSynth_Standalone>/../Info.plist" || true
        COMMENT "Adding App Transport Security exceptions to Standalone App Info.plist"
    )
endif()
